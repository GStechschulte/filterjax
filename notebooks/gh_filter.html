def plot_gh_filter(measurements, filtered_data, title=""):
    plt.figure(figsize=(7, 3))
    plt.plot (filtered_data, "--",label="filter")
    plt.plot(measurements,label="measurement")
    plt.legend(loc=4)
    plt.title(title)
    plt.show()
